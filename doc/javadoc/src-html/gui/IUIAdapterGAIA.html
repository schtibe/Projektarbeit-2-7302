001    package gui;
002    
003    import java.util.List;
004    
005    /**
006     * The adapter interface for a gaia (world)
007     * @param <E> the class to adapt
008     */
009    public interface IUIAdapterGAIA<E>  extends IUIAdapter<E> {
010    
011            /**
012             * Get a list of all IUIAdapterTrafficCarrier
013             * @return a list of all IUIAdapterTrafficCarrier
014             */
015            public List<IUIAdapterTrafficCarrier<?>> getRoads();
016            
017            /**
018             * Get a list of all IUIAdapterVehicle
019             * @return a list of all IUIAdapterVehicle
020             */
021            public List<IUIAdapterVehicle<?>> getVehicles();
022            
023            /**
024             * Get the scale of this gaia
025             * @return the scale of this gaia
026             */
027            public float getScale();
028            
029            /**
030             * Add a new vehicle to the GAIA if the lane allows it
031             * @param mouseY 
032             * @param mouseX 
033             */
034            public void addVehicle(
035                            IUIAdapterLane<?> lane,
036                            int mouseX,
037                            int mouseY
038            ) throws Exception;
039            
040            /**
041             * Destroys the gaia
042             */
043            public void destroy();
044            
045            /**
046             * Get all IUIAdapterWayPoint
047             * @return all IUIAdapterWayPoint
048             */
049            public List<IUIAdapterWayPoint<?>> getWaypoints();
050    }
