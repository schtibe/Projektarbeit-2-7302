001    package gui;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import car.IVehicle;
007    import car.VehicleFactory.VehicleType;
008    
009    import common.GlobalConstants;
010    import common.IVector;
011    import common.Vector;
012    
013    import driver.Character;
014    import driver.Drug;
015    import driver.Physics;
016    import environment.IGaia;
017    import environment.ITrafficCarrier;
018    import environment.IWayPoint;
019    
020    /**
021     * A GAIA adapter implementation 
022     */
023    public class UIAdapterGAIA extends UIAdapter<IGaia>
024            implements IUIAdapterGAIA<IGaia> {
025    
026            protected List<IUIAdapterTrafficCarrier<?>> trafficCarriers = new ArrayList<IUIAdapterTrafficCarrier<?>>();
027            protected List<IUIAdapterVehicle<?>> vehicles = new ArrayList<IUIAdapterVehicle<?>>();
028            protected List<IUIAdapterWayPoint<?>> wayPoints = new ArrayList<IUIAdapterWayPoint<?>>();
029            
030            protected float scale, xMax, yMax;
031            protected IVector maxPos, minPos, correctionVector;
032            
033            /**
034             * Do some calculations to project the simulation to the window
035             * @param mainObject The GAIA
036             * @param xMax The maximal existing x value
037             * @param yMax the maximal existing y value
038             * @throws Exception
039             */
040            public UIAdapterGAIA(IGaia mainObject, float xMax, float yMax) throws Exception {
041                    super(mainObject);
042                    
043                    /*
044                     * Generates the map
045                     */
046                    if(this.mainObject.getTrafficCarriers().isEmpty()) {
047                            throw new Exception("No roads are available.");
048                    }
049                    
050                    //Screen resolution
051                    this.xMax = xMax - GlobalConstants.getInstance().getBorder().getComponent(0) * 2;
052                    this.yMax = yMax - GlobalConstants.getInstance().getBorder().getComponent(1) * 2;
053    
054                                    
055                    for(ITrafficCarrier trafficCarrier : this.mainObject.getTrafficCarriers()) {
056                            this.trafficCarriers.add(UIElementFactory.getUIElement(trafficCarrier, this.scale));
057                    }
058                    
059                    //Get the max and min position of all roads
060                    this.maxPos = this.getMaxPos();
061                    this.minPos = this.getMinPos();
062                    
063                    this.scale = this.calculateScale();
064                    
065                    //scale all and move roads in upper right corner
066                    this.correctionVector = minPos.multiply(-1).multiply(scale).add(new Vector(
067                                    new float[]{
068                                                    GlobalConstants.getInstance().getBorder().getComponent(0),
069                                                    GlobalConstants.getInstance().getBorder().getComponent(1)}));
070                    
071                    for(IUIAdapterTrafficCarrier<?> road : trafficCarriers) {
072                            road.setScale(this.scale);
073                            road.move(correctionVector);
074                    }
075                    
076                    //Get the max and min position of all roads
077                    this.maxPos = this.getMaxPos();
078                    this.minPos = this.getMinPos();
079                    
080                    createVehicles();
081            
082                    createWaypoints();
083            }
084    
085            /**
086             * Create the UI elements for way points
087             * @throws Exception
088             */
089            private void createWaypoints() throws Exception {
090                    for(IWayPoint wayPoint : this.mainObject.getWayPoints()) {
091                            this.wayPoints.add(UIElementFactory.getUIElement(
092                                            wayPoint, 
093                                            this.scale, 
094                                            correctionVector
095                            ));
096                    }
097            }
098    
099            /**
100             * Create the UI elements for vehicles
101             * @throws Exception
102             */
103            private void createVehicles() throws Exception {
104                    for(IVehicle vehicle : this.mainObject.getVehicles()) {
105                            this.vehicles.add(UIElementFactory.getUIElement(
106                                            vehicle, 
107                                            this.scale, 
108                                            correctionVector
109                                    )
110                            );
111                    }
112            }
113    
114            /**
115             * Fits the whole map to one screen
116             * @return the scale factor to fit the map on one screen.
117             */
118            private float calculateScale() {
119                    float xScale, yScale;
120                    
121                    //Try to fit into the screen
122                    xScale = xMax / (this.maxPos.getComponent(0) - this.minPos.getComponent(0));
123                    yScale = yMax / (this.maxPos.getComponent(1) - this.minPos.getComponent(1));
124                    
125                    if(xScale < yScale) {
126                            return xScale;
127                    }
128                    return yScale;
129            }
130            
131            /**
132             * Evaluates the max. positions (Vectors) of all lanes.
133             * @return the max. position
134             */
135            private IVector getMaxPos() {
136                    float[] currentMaxPos = {0,0};
137                    
138                    for(IUIAdapterTrafficCarrier<?> road : this.trafficCarriers){
139                            for(int ix = 0; ix <= 1; ix++){
140                                    if(currentMaxPos[ix] < road.getMaxPos().getComponent(ix)){
141                                            currentMaxPos[ix] = road.getMaxPos().getComponent(ix);
142                                    }
143                            }
144                    }
145                    
146                    return new Vector(currentMaxPos);
147            }
148            
149            /**
150             * Evaluates the min. positions (Vectors) of all lanes.
151             * @return the min. position
152             */
153            private IVector getMinPos() {
154                    float[] currentMinPos = {this.trafficCarriers.get(0).getMinPos().getComponent(0),
155                                                                     this.trafficCarriers.get(0).getMinPos().getComponent(0)};
156                    
157                    for(IUIAdapterTrafficCarrier<?> road : this.trafficCarriers){
158                            for(int ix = 0; ix <= 1; ix++){
159                                    if(currentMinPos[ix] > road.getMinPos().getComponent(ix)){
160                                            currentMinPos[ix] = road.getMinPos().getComponent(ix);
161                                    }
162                            }
163                    }
164                    
165                    return new Vector(currentMinPos);
166            }
167            
168            @Override
169            public List<IUIAdapterTrafficCarrier<?>> getRoads() {
170                    return this.trafficCarriers;
171            }
172            
173            @Override
174            public List<IUIAdapterVehicle<?>> getVehicles() {
175                    return this.vehicles;
176            }
177    
178            @Override
179            public float getScale() {
180                    return this.scale;
181            }
182    
183            @Override
184            public void addVehicle(
185                            IUIAdapterLane<?> lane, 
186                            int mouseX, 
187                            int mouseY) throws Exception {
188                    if (!lane.vehiclePlacable()) {
189                            return;
190                    }
191    
192                    float lanePos = lane.getPositionOnLane(mouseX, mouseY);
193                    
194                    this.vehicles.add(
195                            UIElementFactory.getUIElement(
196                                    this.mainObject.addRoadUser(
197                                            VehicleType.car, 
198                                            lanePos,
199                                            lane.getOriginalLane(),  
200                                            new Character(), 
201                                            new Physics(
202                                                    200,
203                                                    (float)(2*Math.PI-((Math.PI)/3)),
204                                                    250, 
205                                                    new ArrayList<Drug>()
206                                            )
207                                    ),
208                                    this.scale,
209                                    this.correctionVector
210                            )
211                    );
212            }
213    
214            @Override
215            public void destroy() {
216                    this.mainObject.destroy();
217            }
218    
219            @Override
220            public List<IUIAdapterWayPoint<?>> getWaypoints() {
221                    return this.wayPoints;
222            }
223    
224    }
