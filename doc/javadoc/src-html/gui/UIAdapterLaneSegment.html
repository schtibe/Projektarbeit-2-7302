001    package gui;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import common.IVector;
007    
008    import environment.ILaneSegment;
009    
010    /**
011     * A lane segment adapter implementation
012     */
013    public class UIAdapterLaneSegment extends UIAdapter<ILaneSegment<?>> 
014                    implements IUIAdapterLaneSegment<ILaneSegment<?>>{
015            
016            private List<IVector> path ;
017            private static int counter;
018            
019            public UIAdapterLaneSegment(ILaneSegment<?> mainObject) throws Exception {
020                    super(mainObject);
021                    this.path = new ArrayList<IVector>();
022                    this.generatePath();
023                    counter++;
024            }
025            
026            /**
027             * Generates the path
028             */
029            private void generatePath() {
030                    this.path.add(this.mainObject.getStartPoint());
031                    this.path.addAll(this.getBezierList());
032                    this.path.add(this.mainObject.getEndPoint());
033            }
034            
035            /**
036             * Generates the Bezier curve
037             * @return the new bezier curve
038             */
039            private List<IVector> getBezierList() {
040                    return this.getBezierListRecursive(this.mainObject.getStartPoint(),this.mainObject.getEndPoint(),0f,1f);
041            }
042            
043            /**
044             * Generates recursively the Bezier curve 
045             * @return the new Bezier curve
046             */
047            private List<IVector> getBezierListRecursive(IVector start, IVector end, float startFloat, float endFloat){
048                    List<IVector> vectors = new ArrayList<IVector>();
049                    float middleShizzle = (startFloat+endFloat)/2;
050                    
051                    IVector middleVec = this.mainObject.getPointOnCurve(middleShizzle);
052            
053                    if ((middleVec.sub(start).norm()) <= 10f) {
054                            vectors.add(middleVec); 
055                    } else {
056                            vectors.addAll(this.getBezierListRecursive(start, middleVec,startFloat,middleShizzle));
057                            vectors.add(middleVec);
058                            vectors.addAll(this.getBezierListRecursive(middleVec,end, middleShizzle,endFloat));
059                    }
060                    return vectors;
061            }
062            
063            @Override
064            public List<IVector> getPath() {
065                    return this.path;
066            }
067    
068            @Override
069            public IVector getStartPosition() {
070                    return this.mainObject.getStartPoint();
071            }
072    }
