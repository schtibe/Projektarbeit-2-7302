001    package gui;
002    
003    import java.util.List;
004    
005    import org.newdawn.slick.geom.Path;
006    import common.IVector;
007    import environment.ILane;
008    
009    /**
010     * The adapter interface for a lane
011     * @param <E> the class to adapt
012     */
013    public interface IUIAdapterLane<E> extends IUIAdapter<E>, IUIAdapterDrawable {
014    
015            /**
016             * Get the Path
017             * @return the path
018             */
019            public abstract Path getPath();
020            
021            /**
022             * Get the max pos of this object
023             * @return the max pos of this object
024             */
025            public abstract IVector getMaxPos();
026            
027            /**
028             * Get the min pos of this object
029             * @return the min pos of this object
030             */
031            public abstract IVector getMinPos();
032            
033            /**
034             * Set the scale
035             * @scale the new scale factor
036             */
037            public abstract void setScale(float scale) throws Exception;
038            
039            /**
040             * Move the object to this pos
041             * @param pos the new position of this object
042             */
043            public abstract void move(IVector pos);
044            
045            /**
046             * get the object inside of this adapter
047             * @return the object inside of this adapter
048             */
049            public ILane getOriginalLane();
050    
051            /**
052             * Return whether a vehicle can be placed on this lane
053             * 
054             * Vehicles cannot be placed on some lanes, e.g. on those
055             * that are in junctions. Return here if it is possible to
056             * place a vehicle on this lane.
057             * @return
058             */
059            public abstract boolean vehiclePlacable();
060            
061            /**
062             * Return the position on the lane
063             * @param mouseX The x position of the mouse
064             * @param mouseY The y position of the mouse
065             * @return Position on lane
066             */
067            public float getPositionOnLane(int mouseX, int mouseY);
068            
069            /**
070             * REturn the paths of the lane segments
071             * @return Lane segment paths
072             */
073            public List<Path> getLaneSegmentPaths();
074    }
