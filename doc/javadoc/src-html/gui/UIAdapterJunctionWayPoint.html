001    package gui;
002    
003    import org.newdawn.slick.Color;
004    import org.newdawn.slick.Graphics;
005    import org.newdawn.slick.TrueTypeFont;
006    import org.newdawn.slick.geom.Rectangle;
007    import org.newdawn.slick.geom.Shape;
008    
009    import common.IVector;
010    
011    import environment.IWayPoint;
012    
013    /**
014     * A way point adapter implementation
015     */
016    public class UIAdapterJunctionWayPoint extends UIAdapter<IWayPoint> implements IUIAdapterJunctionWayPoint {
017            /**
018             * A shape which represents the way point
019             */
020            protected Shape shape;
021            
022            public UIAdapterJunctionWayPoint(
023                            IWayPoint mainObject, 
024                            float scale, 
025                            IVector offsetVector
026            ) throws Exception {
027                    super(mainObject);
028                    this.shape = new Rectangle(
029                                    0, 
030                                    0, 
031                                    10f, 
032                                    10f
033                    );
034                    
035                    this.shape.setCenterX(this.mainObject.getXPos() * scale + offsetVector.getComponent(0));
036                    this.shape.setCenterY(this.mainObject.getYPos() * scale + offsetVector.getComponent(1)); 
037            }
038    
039            @Override
040            public Shape getShape() {
041                    return this.shape;
042            }
043            
044            @Override
045            public String toString() {
046                    return this.mainObject.toString();
047            }
048            
049            /**
050             * Draw the string 
051             * @param ttf The font tool
052             */
053            private void drawString(TrueTypeFont ttf) {
054                    if (GUIConstants.getInstance().showJunctionWaypointPosition()) {
055                            ttf.drawString(this.getShape().getCenterX(),
056                                               this.getShape().getCenterY(), 
057                                               this.toString(),Color.orange);
058                    }
059            }
060            
061            @Override
062            public boolean doDraw() {
063                    return true;
064            }
065    
066            @Override
067            public void draw(Graphics g, TrueTypeFont ttf) {
068                    if (this.doDraw()) {
069                            g.setColor(Color.white);
070                            g.draw(this.getShape());
071                            this.drawString(ttf);
072                    }
073            
074            }
075            
076    
077    }
