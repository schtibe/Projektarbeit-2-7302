001    package gui;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import org.newdawn.slick.GameContainer;
007    import org.newdawn.slick.Graphics;
008    import org.newdawn.slick.Input;
009    import org.newdawn.slick.SlickException;
010    import org.newdawn.slick.opengl.SlickCallable;
011    import org.newdawn.slick.state.BasicGameState;
012    import org.newdawn.slick.state.StateBasedGame;
013    
014    import de.lessvoid.nifty.EndNotify;
015    import de.lessvoid.nifty.Nifty;
016    import de.lessvoid.nifty.effects.EffectEventId;
017    import de.lessvoid.nifty.elements.Element;
018    import de.lessvoid.nifty.input.mouse.MouseInputEvent;
019    import de.lessvoid.nifty.lwjglslick.render.RenderDeviceLwjgl;
020    import de.lessvoid.nifty.lwjglslick.sound.SlickSoundDevice;
021    import de.lessvoid.nifty.screen.Screen;
022    import de.lessvoid.nifty.screen.ScreenController;
023    import de.lessvoid.nifty.sound.SoundSystem;
024    import de.lessvoid.nifty.spi.input.InputSystem;
025    import de.lessvoid.nifty.tools.TimeProvider;
026    
027    /**
028     * A game state to represent the main menu
029     *
030     */
031    public class GameStateMainMenu extends BasicGameState implements
032                    ScreenController {
033            public static final int ID = 0;
034            private GameContainer container;
035            private Nifty nifty;
036            private List<MouseInputEvent> mouseEvents = new ArrayList<MouseInputEvent>();
037            private int mouseX;
038            private int mouseY;
039            private boolean mouseDown;
040            private boolean showFPS = false;
041            private StateBasedGame game;
042            private Element errorBox;
043    
044            @Override
045            public int getID() {
046                    return ID;
047            }
048    
049            @Override
050            public void init(GameContainer container, StateBasedGame game)
051                            throws SlickException {
052                    this.container = container;
053                    this.game = game;
054    
055                    // create nifty (gui)
056                    nifty = new Nifty(new RenderDeviceLwjgl(), new SoundSystem(
057                                    new SlickSoundDevice()), new InputSystem() {
058                            @Override
059                            public List<MouseInputEvent> getMouseEvents() {
060                                    ArrayList<MouseInputEvent> result = new ArrayList<MouseInputEvent>(
061                                                    mouseEvents);
062                                    mouseEvents.clear();
063                                    return result;
064                            }
065                    }, new TimeProvider());
066                    nifty.fromXml("ressources/gui/mainMenu.xml", "mainMenu", this);
067    
068                    // fetch the error box from the xml
069                    this.errorBox = getElement("errorLayer");
070    
071            }
072    
073            @Override
074            public void render(GameContainer container, StateBasedGame game, Graphics g) {
075                    // draw the gui
076                    SlickCallable.enterSafeBlock();
077                    nifty.render(false);
078                    SlickCallable.leaveSafeBlock();
079            }
080    
081            @Override
082            public void update(GameContainer container, StateBasedGame game, int delta) {
083            }
084    
085            @Override
086            public void keyReleased(int key, char c) {
087                    // toggle fps
088                    if (key == Input.KEY_F) {
089                            this.showFPS = !this.showFPS;
090                    }
091            }
092    
093            private Element getElement(final String id) {
094                    return nifty.getCurrentScreen().findElementByName(id);
095            }
096    
097            @Override
098            public void mouseMoved(final int oldx, final int oldy, final int newx,
099                            final int newy) {
100                    mouseX = newx;
101                    mouseY = newy;
102                    forwardMouseEventToNifty(mouseX, mouseY, mouseDown);
103            }
104    
105            @Override
106            public void mousePressed(final int button, final int x, final int y) {
107                    mouseX = x;
108                    mouseY = y;
109                    mouseDown = true;
110                    forwardMouseEventToNifty(mouseX, mouseY, mouseDown);
111            }
112    
113            @Override
114            public void mouseReleased(final int button, final int x, final int y) {
115                    mouseX = x;
116                    mouseY = y;
117                    mouseDown = false;
118                    forwardMouseEventToNifty(mouseX, mouseY, mouseDown);
119            }
120    
121            private void forwardMouseEventToNifty(final int mouseX, final int mouseY,
122                            final boolean mouseDown) {
123                    mouseEvents.add(new MouseInputEvent(mouseX, container.getHeight()
124                                    - mouseY, mouseDown));
125            }
126    
127            @Override
128            public void bind(Nifty nifty, Screen screen) {
129            }
130    
131            @Override
132            public void onEndScreen() {
133            }
134    
135            @Override
136            public void onStartScreen() {
137            }
138    
139            /**
140             * Quit
141             */
142            public void quit() {
143                    nifty.getCurrentScreen().endScreen(new EndNotify() {
144                            @Override
145                            public void perform() {
146                                    container.exit();
147                            }
148                    });
149            }
150    
151            /**
152             * Start the simulation
153             * @throws SlickException
154             */
155            public void startSimulation() throws SlickException {
156                    game.enterState(3);
157            }
158    
159            /**
160             * Apply the choice of resolution
161             */
162            public void startChangeResolution() {
163                    this.errorBox.startEffect(EffectEventId.onCustom);
164                    UIElementErrorMessageBox.create(nifty, getElement("errorLayer"),
165                                    getElement("mainPanel"), "help!");
166            }
167    
168            /**
169             * 
170             */
171            public void startChangeMap() {
172                    game.enterState(2);
173            }
174    }
