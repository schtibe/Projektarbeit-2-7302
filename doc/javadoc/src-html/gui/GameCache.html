001    package gui;
002    
003    /**
004     * Holds information which are relevant for the gui
005     */
006    public class GameCache {
007            private static GameCache instance;
008            private IUIAdapterGAIA<?> GAIA;
009            
010            public synchronized static GameCache getInstance() {
011            if (instance == null) {
012                instance = new GameCache();
013            }
014            return instance;
015        }
016            
017            /**
018             * stores one gaia
019             */
020            public void setGAIA(IUIAdapterGAIA<?> GAIA) {
021                    this.GAIA = GAIA;
022            }
023            
024            /**
025             * Get the gaia adapter
026             */
027            public IUIAdapterGAIA<?> getGAIA() {
028                    return this.GAIA;
029            }
030    }
