001    package gui;
002    
003    import de.lessvoid.nifty.Nifty;
004    import de.lessvoid.nifty.controls.dynamic.CustomControlCreator;
005    import de.lessvoid.nifty.controls.dynamic.LabelCreator;
006    import de.lessvoid.nifty.controls.dynamic.PanelCreator;
007    import de.lessvoid.nifty.effects.EffectEventId;
008    import de.lessvoid.nifty.elements.Element;
009    
010    /**
011     * A small error dialog using nifty
012     */
013    public class UIElementErrorMessageBox {
014            public static void create(Nifty nifty, Element show, Element hide, String errorMessage) {               
015                    //Red error panel
016                    PanelCreator errorPanel = new PanelCreator("errorPanelID");
017                    errorPanel.setChildLayout("vertical");
018                    errorPanel.setStyle("nifty-panel-red");
019                    errorPanel.setVAlign("center");
020                    errorPanel.setAlign("center");
021                    errorPanel.setHeight("150px");
022                    errorPanel.setWidth("400px");
023                    errorPanel.setVisible("true");
024                    Element Element1 = errorPanel.create(nifty,  nifty.getCurrentScreen(), show);
025    
026                    //Error label
027                LabelCreator errorLabel = new LabelCreator("errorLabelID","Error");
028                errorLabel.setAlign("center");
029                errorLabel.setVAlign("center");
030                errorLabel.setStyle("nifty-label");
031               
032                errorLabel.create(nifty, nifty.getCurrentScreen(), nifty.getCurrentScreen().findElementByName("errorPanelID"));
033                
034                    //Error message
035                LabelCreator errorMessageLabel = new LabelCreator("errorMessageID",errorMessage);
036                errorMessageLabel.setAlign("center");
037                errorMessageLabel.setVAlign("center");
038                errorMessageLabel.setStyle("nifty-label");
039                errorMessageLabel.create(nifty, nifty.getCurrentScreen(), nifty.getCurrentScreen().findElementByName("errorPanelID"));
040                
041                //Error button
042                CustomControlCreator errorButton = new CustomControlCreator("errorButton","button");
043                errorButton.setAlign("center");
044                errorButton.setVAlign("center");
045                errorButton.set("label", "exit");
046                errorButton.setInteractOnClick("quit()");
047                errorButton.create(nifty, nifty.getCurrentScreen(), nifty.getCurrentScreen().findElementByName("errorPanelID"));
048                
049                    //Hide GUI
050                    //hide.disable();
051                hide.startEffect(EffectEventId.onCustom);
052                /*
053                for(Element e1 : hide.getElements()) {
054                    //e1.setFocusable(false);
055                    //e1.setVisible(false);
056                    //e1.disable();
057                }
058                */
059                    Element1.setVisibleToMouseEvents(true);
060            }
061    }
