001    package gui;
002    
003    /**
004     * Some properties for the gui
005     */
006    public class GUIConstants {
007            
008            /**
009             * Singleton instance
010             */
011            private static GUIConstants instance;
012            
013            public synchronized static GUIConstants getInstance() {
014                    if (GUIConstants.instance == null) {
015                            GUIConstants.instance = new GUIConstants();
016                    }
017                    
018                    return GUIConstants.instance;
019            }
020            
021            /**
022             * Private constructor
023             */
024            private GUIConstants() {}
025            
026            /**
027             * Whether the position of the junction way points are to be shown
028             */
029            private boolean showJunctionWaypointPosition = false;
030            public boolean showJunctionWaypointPosition() {
031                    return this.showJunctionWaypointPosition;
032            }
033            public boolean toggleShowJunctionWaypointPosition() {
034                    this.showJunctionWaypointPosition = !this.showJunctionWaypointPosition;
035                    return this.showJunctionWaypointPosition;
036            }
037            
038            /**
039             * Whether the position of the speed way points are to be shown
040             */
041            private boolean showSpeedWaypointPosition = true;
042            public boolean showSpeedWaypointPosition() {
043                    return this.showSpeedWaypointPosition;
044            }
045            public boolean toggleShowSpeedWaypointPosition() {
046                    this.showSpeedWaypointPosition = !this.showSpeedWaypointPosition;
047                    return this.showSpeedWaypointPosition;
048            }
049    
050            /**
051             * Whether the driver view is to be shown
052             */
053            private boolean showDriverView = true;
054            public boolean showDriverView() {
055                    return this.showDriverView;
056            }
057            public boolean toggleShowDriverView() {
058                    this.showDriverView = !this.showDriverView;
059                    return this.showDriverView;
060            }
061    }
