001    package environment;
002    
003    import common.IVector;
004    
005    public abstract class MovingWayPoint extends WayPoint implements IMovable {
006    
007            protected IVector position;
008            
009            /**
010             * Construct
011             * @param IVector position
012             */
013            
014            public MovingWayPoint(IVector position) {
015                    this.position = position;
016            }
017    
018            @Override
019            public float getXPos() {
020                    return this.position.getComponent(0);
021            }
022            
023            @Override
024            public float getYPos() {
025                    return this.position.getComponent(1);
026            }
027    
028            @Override
029            public void updatePosition(IVector pos) {
030                    this.position = pos;
031            }
032            
033            @Override
034            public String toString(){
035                    return "type: moving\ncoordinates: "+this.position.toString();
036            }
037            
038            @Override
039            public IVector getPosition() {
040                    return this.position;
041            }
042    }
