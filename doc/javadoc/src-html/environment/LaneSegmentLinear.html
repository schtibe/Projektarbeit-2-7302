001    package environment;
002    
003    import common.IVector;
004    
005    
006    public class LaneSegmentLinear extends LaneSegment<ILaneSegmentQuadratic> 
007                    implements ILaneSegmentLinear {
008    
009            /**
010             * this is the constructor
011             * @param id
012             * @param startPoint
013             * @param endPoint
014             * @throws Exception
015             */
016            public LaneSegmentLinear(int id, IVector startPoint, IVector endPoint) throws Exception{
017                    super(startPoint, endPoint);
018                    this.length = calculateLength();
019                    this.id = id;
020            }
021    
022            @Override
023            protected float calculateLength() {
024                    return relativeEndPoint.norm();
025            }
026    
027            @Override
028            public IVector getVehiclePosition(float segmentLength) {
029                    float time = segmentLength/this.length;
030                    return getPointOnCurve(time);
031            }
032            
033            @Override
034            public IVector getPointOnCurve(float time){
035                    if (time >= 0f && time <= 1f){
036                            IVector result = relativeEndPoint.multiply(time);
037                            return result.add(this.startPoint);
038                    }else{
039                            return this.endPoint;
040                    }
041            }
042            
043            @Override
044            public IVector[] getBezierPoints() {
045                    IVector[] output = new IVector[4];
046                    output[0] = this.startPoint;
047                    output[1] = output[2] = output[3] = this.endPoint;
048                    return output;
049            }
050    
051            @Override
052            public float positionIntersection(IVector position) throws Exception {
053                    return this.getDistanceOnLane();
054            }
055    }
