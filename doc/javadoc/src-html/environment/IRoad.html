001    package environment;
002    
003    import java.util.List;
004    
005    import common.IVector;
006    
007    public interface IRoad extends ITrafficCarrier{
008            /**
009             * Set the left lanes
010             * 
011             * @param leftLanes the leftLanes to set
012             */
013            public void setLeftLanes(List<ILane> leftLanes);
014            
015            /**
016             * Set the right lanes
017             * @param rightLanes
018             */
019            public void setRightLanes(List<ILane> rightLanes);
020            
021            
022            
023            /**
024             * Set the start point of the road
025             * 
026             * @param startPoint the startPoint to set
027             */
028            public void setStartPoint(IVector startPoint);
029            
030            /**
031             * Set the start point
032             * 
033             * @param endPoint the endPoint to set (relative to the startPoint)
034             */
035            public void setEndPoint(IVector endPoint);
036            
037            /**
038             * Return the end point
039             * 
040             * @return the endPoint (relative to the startPoint)
041             */
042            public IVector getEndPoint();
043            
044            /**
045             * Return the start point
046             * 
047             * @return the startPoint
048             */
049            public IVector getStartPoint();
050    }
