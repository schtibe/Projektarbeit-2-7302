001    package environment;
002    
003    import driver.IDirection;

005    public interface IPriority {

007             * returns if the other car has priority over the own car
008             * @param dir
009             * @param from
010             * @param to
011             * @return true if other has priority false else
012             */
013            
014            boolean hasPriority (IDirection dir, IDirection from, IDirection to);
015            
016            /**
017             * call back for visitor on direction
018             * @param dir
019             */
020            
021            void handleDir (IDirection dir);
022            
023            /**
024             * call back for visitor on direction
025             * @param dir
026             */
027            
028            void handleFrom (IDirection dir);
029            
030            /**
031             * call back for visitor on direction
032             * @param dir
033             */
034            
035            void handleTo (IDirection dir);
036    }
