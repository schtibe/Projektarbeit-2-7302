001    package environment;
002    
003    import java.util.List;
004    
005    import driver.IDirection;
006    
007    public interface IJunction extends ITrafficCarrier{
008            
009            /**
010             * set the roads that go together in this junction
011             * @param roads
012             * @throws Exception
013             */
014            
015            void setRoads (List<IRoad> roads) throws Exception;
016            
017            /**
018             * get the next possible lanes
019             * @param actualLane
020             * @return
021             */
022            
023            List<IJunctionDecision> getPossibilities(ILane actualLane);
024            
025            /**
026             * get the direction of the other vehicle according to which lane I am on
027             * @param actualLane
028             * @param otherLane
029             * @return
030             */
031            
032            IDirection comingFrom (ILane actualLane, ILane otherLane);
033            
034            /**
035             * returns the list of all the other lanes running through the junction
036             * @param actualLane
037             * @return List<ILane>
038             */
039            
040            List<ILane> getRelevantLanes (ILane actualLane);
041    
042    }
