046            @Override
047            public void setRoads(List<IRoad> roads) throws Exception{
048                    this.roads = roads;
049                    roadsToLanes();
050                    LaneBuilder builder = new LaneBuilder();        
051                    int decisionCount = 0;
052                    for (int i=0;i<this.incomingLanes.length;i++){
053                            if(this.incomingLanes[i]!=null){
054                                    for (ILane lane: this.incomingLanes[i]) {
055                                            if (!this.decisionsMap.contains(lane)){
056                                                    this.decisionsMap.add(lane);
057                                            }
058                                            int laneIdx = this.decisionsMap.indexOf(lane);
059                                            LaneSegmentLinear start = (LaneSegmentLinear) lane.getLastILaneSegment();
060                                            for (int k=0;k<this.outgoingLanes.length;k++){
061                                                    if (this.outgoingLanes[k]!=null && k!= i){
062                                                            for (ILane outLane: this.outgoingLanes[k]) {
063                                                                    if (outLane != null) {
064                                                                            List<ILane> theseLanes = makeJunctionLanes(
065                                                                                            builder, start, outLane);
066                                                                            decisionCount++;
067                                                                            IDirection dir = this.getDirectionForDecision(lane, outLane);
068                                                                            IJunctionDecision decision = new JunctionDecision(theseLanes,decisionCount,this,dir);
069                                                                            try{
070                                                                                    decisions.get(laneIdx);
071                                                                            }catch(Exception e){
072                                                                                    decisions.add(laneIdx,new ArrayList<IJunctionDecision>());
073                                                                            }
074                                                                            //makeJunctionWayPoint(lane);
075                                                                            decisions.get(laneIdx).add(decision);
076                                                                    }
077                                                            }
078                                                    }
079                                            }
080                                            makeJunctionWayPoint(lane);
081                                    }
082                            }
083                    }
084            }

086            /**
087             * attributes a Direction object to a certain combination of lanes
088             * @param coming
089             * @param going
090             * @return IDirection instance
091             */
092            
093            private IDirection getDirectionForDecision (ILane coming,ILane going){
094                    IVector incomingStart = coming.getLastILaneSegment().getStartPoint();
095                    IVector incomingEnd = coming.getLastILaneSegment().getEndPoint();
096                    IVector goingStart = going.getFirstILaneSegment().getStartPoint();
097                    IVector goingEnd = going.getLastILaneSegment().getEndPoint();
098                    IVector turnDirection;
099                    if (goingStart.sub(incomingEnd).norm()<goingEnd.sub(incomingEnd).norm()){
100                            turnDirection = going.getFirstILaneSegment().getEndPoint().sub(goingStart);
101                    }else{
102                            turnDirection = going.getLastILaneSegment().getStartPoint().sub(goingEnd);
103                    }
104                    IVector incomingDirection = incomingEnd.sub(incomingStart);
105                    //System.out.println("in:"+incomingDirection.getAngle()+";out:"+turnDirection.getAngle());
106                    //float angle = incomingDirection.getAngle()-turnDirection.getAngle();
107                    float angle = incomingDirection.getAngle()-turnDirection.getAngle();
108                    if (angle < 0){
109                            angle += 2*Math.PI;
110                    }
111                    //System.out.println("angle:"+(angle/Math.PI)*180);
112                    if (angle <= Math.PI/8 || angle >= 15*Math.PI/8){
113                            return new StraightDirection();
114                    }else if (angle < 15*Math.PI/8 && angle > Math.PI){
115                            return new RightDirection();
116                    }else{
117                            return new LeftDirection();
118                    }
119            }

131            /**
132             * creates the junction lanes for an incoming lane
133             * @param builder
134             * @param start
135             * @param outLane
136             * @return the list of the lanes in the junction
137             * @throws Exception
138             * @throws InvalidXMLException
139             */
140            private List<ILane> makeJunctionLanes(LaneBuilder builder,
141                            LaneSegmentLinear start, ILane outLane) throws Exception,
142                            InvalidXMLException {
143                    
144                    outLane.setJunction(this);
145                    LaneSegmentLinear end = (LaneSegmentLinear) outLane.getFirstILaneSegment();
146                    IVector startVector   = (start.getEndPoint().sub(start.getStartPoint())).normalize();
147                    IVector endVector     = (end.getEndPoint().sub(end.getStartPoint())).normalize();
148                    List<ILaneSegmentLinear> segments = new ArrayList<ILaneSegmentLinear>();
149                    
150                    segments.add(new LaneSegmentLinear(1,start.getEndPoint(),((Vector)start.getEndPoint()).add(startVector)));
151                    segments.add(new LaneSegmentLinear(1,((Vector) end.getStartPoint()).sub(endVector), end.getStartPoint()));
152                    
153                    Lane newLane = null;
154                    try {
155                            newLane = new Lane(start.getEndPoint(),builder.getCompleteLaneSegmentList(segments),(float)10.0, false);
156                    } catch(InvalidXMLException e) {
157                            segments = new ArrayList<ILaneSegmentLinear>();
158                            segments.add(new LaneSegmentLinear(1,start.getEndPoint(),end.getStartPoint()));
159                            newLane = new Lane(start.getEndPoint(),builder.getCompleteLaneSegmentList(segments),(float)10.0, false);
160                    }
161            
162                    this.junctionLanes.add(newLane);
163                    List<ILane> theseLanes = new ArrayList<ILane>();
164                    theseLanes.add(newLane);
165                    theseLanes.add(outLane);
166                    return theseLanes;
167            }

169            /**
170             * separates the incoming from the outgoing lanes
171             * @throws Exception
172             */
173            private void roadsToLanes() throws Exception {
174                    Object[] roadsAsArray = this.roads.toArray();
175                    startPointInCrossroad = findPointsInCrossroad(roadsAsArray);
176                    this.incomingLanes = new List[startPointInCrossroad.length];
177                    this.outgoingLanes = new List[startPointInCrossroad.length];
178                    for (int i=0;i<startPointInCrossroad.length;i++){
179                            if (startPointInCrossroad[i]){
180                                    this.incomingLanes[i] = ((ITrafficCarrier)roadsAsArray[i]).getLeftLanes();
181                                    this.outgoingLanes[i] = ((ITrafficCarrier)roadsAsArray[i]).getRightLanes();
182                            }else{
183                                    this.incomingLanes[i] = ((ITrafficCarrier)roadsAsArray[i]).getRightLanes();
184                                    this.outgoingLanes[i] = ((ITrafficCarrier)roadsAsArray[i]).getLeftLanes();
185                            }
186                    }
187            }

189            /**
190             * true means the startPoint of the Road with index i is to be connected to the cross roads, 
191             * false means the same for the endPoint
192             * @return Boolean[]
193             * 
194             */
195            
196            private Boolean[] findPointsInCrossroad(Object[] roadsAsArray) throws Exception{
197                    if (roadsAsArray.length > 1 || roadsAsArray.length > 4) {
198                            Boolean[] output = new Boolean[roadsAsArray.length];
199                            for (int i = 0; i < roadsAsArray.length; i++) {
200                                    if (i == 0) {
201                                            // stay idle
202    
203                                    }else if(i == 1){
204                                            firstTwoPointsInCrossroad((IRoad)roadsAsArray[0], (IRoad)roadsAsArray[1],output);
205                                    }else {
206                                            if (output[i - 1]) {
207                                                    float toStart = ((IRoad)roadsAsArray[(i - 1)])
208                                                                    .getStartPoint().sub(
209                                                                                    ((IRoad)roadsAsArray[i]).getStartPoint()).norm();
210                                                    float toEnd = ((IRoad)roadsAsArray[(i - 1)])
211                                                                    .getStartPoint().sub(
212                                                                                    ((IRoad)roadsAsArray[i]).getEndPoint()).norm();
213                                                    output[i] = endOrStart(toEnd,toStart);
214                                            } else {
215                                                    float toEnd = ((IRoad)roadsAsArray[(i - 1)]).getEndPoint()
216                                                                    .sub(((IRoad)roadsAsArray[i]).getEndPoint()).norm();
217                                                    float toStart = ((IRoad)roadsAsArray[(i - 1)]).getEndPoint()
218                                                                    .sub(((IRoad)roadsAsArray[i]).getStartPoint()).norm();
219                                                    output[i] = endOrStart(toEnd,toStart);
220                                            }
221                                    }
222                            }
223                            return output;
224                    }else{
225                            throw new Exception ("a crossroad needs more than one road and less than five");
226                    }
227            }

229            /**
230             * @param length to a neighbouring endPoint,
231             * @param length to a neighbouring startPoint
232             * @return true if toStart is larger than toEnd
233             */
234            
235            private Boolean endOrStart(float toEnd, float toStart){
236                    if (toEnd < toStart){
237                            return false;
238                    }else{
239                            return true;
240                    }
241            }

243            /**
244             * finds the first two points of a crossroad according to the provided roads
245             * @param roadOne
246             * @param roadTwo
247             * @param ouput, contains the values accoridng to spec in findPointsInCrossroad
248             * @TODO beautify: better return an array than fill it in the function
249             */
250            
251            private void firstTwoPointsInCrossroad(IRoad roadOne, IRoad roadTwo,Boolean[] output) {
252                    //Boolean[] output = new Boolean[2];
253                    float endToEnd = roadOne.getEndPoint().sub(roadTwo.getEndPoint()).norm();
254                    float endToStart = roadOne.getEndPoint().sub(roadTwo.getStartPoint()).norm();
255                    float startToEnd = roadOne.getStartPoint().sub(roadTwo.getEndPoint()).norm();
256                    float startToStart = roadOne.getStartPoint().sub(roadTwo.getStartPoint()).norm();
257                    if (endToEnd < startToStart) {
258                            if (endToEnd < startToEnd) {
259                                    if (endToEnd < endToStart) {
260                                            output[0] = false;
261                                            output[1] = false;
262                                    } else {
263                                            output[0] = false;
264                                            output[1] = true;
265                                    }
266                            } else {
267                                    if (startToEnd < endToStart) {
268                                            output[0] = true;
269                                            output[1] = false;
270                                    } else {
271                                            output[0] = false;
272                                            output[1] = true;
273                                    }
274                            }
275                    } else {
276                            if (startToStart < startToEnd) {
277                                    if (startToStart < endToStart) {
278                                            output[0] = true;
279                                            output[1] = true;
280                                    } else {
281                                            output[0] = false;
282                                            output[1] = true;
283                                    }
284                            } else {
285                                    if (startToEnd < endToStart) {
286                                            output[0] = true;
287                                            output[1] = false;
288                                    } else {
289                                            output[0] = false;
290                                            output[1] = true;
291                                    }
292                            }
293                    }
294            }

416            @Override
417            public List<ILane> getRelevantLanes(ILane actualLane) {
418                    Set<ILane> relevant = new HashSet();
419                    relevant.addAll(this.junctionLanes);
420                    for (List<ILane> lanes :incomingLanes){
421                            if (!lanes.contains(actualLane)){
422                                    relevant.addAll(lanes);
423                            }
424                    }
425                    List<ILane> output = new ArrayList<ILane>(relevant);
426                    return output;
427            }
