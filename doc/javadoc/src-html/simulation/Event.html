001    package simulation;
002    
003    /**
004     * The abstract event
005     * 
006     * @param <E>
007     *            The real type of the event
008     */
009    public abstract class Event<E> implements IEvent {
010            /**
011             * The time stamp of the event
012             */
013            protected long timestamp;
014    
015            /**
016             * The target to execute the event on
017             */
018            protected IEventTarget<E> target;
019    
020            /**
021             * Initialise
022             * 
023             * @param timestamp
024             * @param target
025             */
026            public Event(long timestamp, IEventTarget<E> target) {
027                    this.timestamp = timestamp;
028                    this.target = target;
029            }
030    
031            @Override
032            public long getTimeStamp() {
033                    return this.timestamp;
034            }
035    
036    
037            @Override
038            public IEventTarget<E> getTarget() {
039                    return this.target;
040            }
041    
042            @Override
043            public int compareTo(IEvent arg0) {
044                    return (int) (this.timestamp - arg0.getTimeStamp());
045            }
046    
047            @Override
048            public abstract void handleEvent() throws Exception;
049    }
