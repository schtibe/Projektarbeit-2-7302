001    package simulation;
002    
003    /**
004     * Interface for event targets
005     * 
006     * Makes them implement a handle event method according to the given event type
007     * 
008     * @param <E> The type of Event that the target expects
009     */
010    public interface IEventTarget<E> {
011            /**
012             * Handle the event
013             * 
014             * @param event
015             * @throws Exception
016             */
017            public void handleEvent(E event) throws Exception;
018    }
