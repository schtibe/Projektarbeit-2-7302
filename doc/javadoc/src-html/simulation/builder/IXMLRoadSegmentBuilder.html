001    package simulation.builder;
002    
003    import simulation.builder.XMLLaneBuilder.directionType;
004    
005    import common.IVector;
006    
007    import environment.ILaneSegmentLinear;
008    
009    /**
010     * Factory class for the road segments
011     */
012    public interface IXMLRoadSegmentBuilder extends
013                    Comparable<IXMLRoadSegmentBuilder> {
014            /**
015             * @param endPoint
016             *            the endPoint to set
017             */
018            public void setEndPoint(IVector endPoint);
019    
020            /**
021             * Get the end point
022             * 
023             * @return the endPoint
024             */
025            public IVector getEndPoint();
026    
027            /**
028             * Set the start point
029             * 
030             * @param startPoint
031             *            the startPoint to set
032             */
033            public void setStartPoint(IVector startPoint);
034    
035            /**
036             * The start point
037             * 
038             * @return the startPoint
039             */
040            public IVector getStartPoint();
041    
042            /**
043             * Return a lane segment
044             * 
045             * @param offset
046             * @return
047             * @throws Exception
048             */
049            public ILaneSegmentLinear createLaneSegment(float offset,
050                            directionType direction) throws Exception;
051    
052            /**
053             * Return the order of the road segment
054             * 
055             * @return
056             */
057            public abstract int getOrder();
058    }
