001    package simulation.builder;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import org.jdom.Element;
007    
008    import common.IVector;
009    
010    import environment.ILaneSegmentLinear;
011    import environment.Lane;
012    
013    /**
014     * Abstract class for lane builders
015     * 
016     * @TODO there should probably be an interface
017     */
018    public abstract class XMLLaneBuilder extends LaneBuilder implements
019                    IXMLObjectBuilder {
020            /**
021             * The available directions for the factory
022             */
023            public enum directionType {
024                    right, left
025            }
026    
027            /**
028             * The direction this lane has
029             */
030            protected directionType direction = directionType.right;
031    
032            /**
033             * The width of the lane
034             */
035            protected float laneWidth = 1;
036    
037            /**
038             * The position of the overlying road
039             */
040            protected IVector roadPosition;
041    
042            /**
043             * The XML element
044             */
045            protected Element elem;
046    
047            /**
048             * Construct
049             * 
050             * @param e The XML element
051             * @param roadPosition
052             */
053            public XMLLaneBuilder(Element e, IVector roadPosition) {
054                    this.elem = e;
055                    this.roadPosition = roadPosition.clone();
056    
057                    this.laneWidth = Float.parseFloat(this.elem.getAttributeValue("width"));
058            }
059    
060            /**
061             * @param laneWidth
062             *            the laneWidth to set
063             */
064            public void setLaneWidth(float laneWidth) {
065                    this.laneWidth = laneWidth;
066            }
067    
068            /**
069             * @return the laneWidth
070             */
071            public float getLaneWidth() {
072                    return laneWidth;
073            }
074    
075            /**
076             * Create and return the lane
077             * 
078             * @param roadSegments
079             * @return
080             * @throws IllegalArgumentException
081             * @throws InvalidXMLException
082             */
083            public Lane createLane(List<IXMLRoadSegmentBuilder> roadSegments,
084                            int counter) throws IllegalArgumentException, InvalidXMLException {
085                    List<ILaneSegmentLinear> laneSegments = new ArrayList<ILaneSegmentLinear>();
086    
087                    if (roadSegments.size() == 0) {
088                            throw new IllegalArgumentException("received no road segments");
089                    }
090    
091                    // build all the linear right lane segments
092                    for (IXMLRoadSegmentBuilder segment : roadSegments) {
093                            try {
094                                    laneSegments.add(
095                                            segment.createLaneSegment(
096                                                    counter + this.laneWidth / 2, 
097                                                    this.direction
098                                            )
099                                    );
100                            } catch (Exception e) {
101                                    e.printStackTrace();
102                            }
103                    }
104                    // this lane has to have the start point of the road
105                    ILaneSegmentLinear startSegment = this
106                                    .getCompleteLaneSegmentList(laneSegments);
107                    
108                    Lane lane = new Lane(
109                            startSegment.getStartPoint().clone(), 
110                            startSegment,
111                            this.laneWidth,
112                            true
113                    );
114                    
115                    for (IXMLWayPointBuilder wp: this.readWayPoints()) {
116                            lane.addWayPoint(wp.createWayPoint(lane));
117                    }
118                    
119                    return lane;
120            }
121    
122            /**
123             * {@inheritDoc}
124             * 
125             * @return
126             * @throws InvalidXMLException 
127             */
128            public List<IXMLWayPointBuilder> readWayPoints() throws InvalidXMLException {
129                    List<IXMLWayPointBuilder> wayPoints = new ArrayList<IXMLWayPointBuilder>();
130                    List<?> wayPointElems = new ArrayList<Element>();
131    
132                    try {
133                            wayPointElems = this.elem.getChild("waypoints").getChildren();
134                    } catch (NullPointerException e) {
135                            // if there are no waypoints - doesn't matter
136                    }
137    
138                    for (int i = 0; i < wayPointElems.size(); i++) {
139                            wayPoints.add(
140                                    XMLWayPointBuilderFactory.getWayPointBuilder(
141                                            (Element) wayPointElems.get(i)
142                                    )
143                            );
144                    }
145                    return wayPoints;
146            }
147    }
