001    package simulation.builder;
002    
003    import org.jdom.Element;
004    
005    /**
006     * Way point factory
007     */
008    public class XMLWayPointBuilderFactory {
009            /**
010             * The possible directions for way points
011             */
012            public enum direction {
013                    right, left, both
014            }
015    
016            /**
017             * Create the accordant way point
018             * 
019             * @param elem
020             * @param wBuilder
021             * @return
022             * @throws InvalidXMLException 
023             */
024            public static IXMLWayPointBuilder getWayPointBuilder(Element elem) 
025                            throws InvalidXMLException {
026                    
027                    String name = elem.getName();
028                    IXMLWayPointBuilder wayPoint = null;
029                    if (name.equals("SpeedWaypoint")) {
030                            wayPoint = new XMLSpeedWayPointBuilder(elem);
031                    }
032    
033                    return wayPoint;
034            }
035    }
