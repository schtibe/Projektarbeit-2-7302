001    package simulation;
002    
003    import car.Vehicle;
004    
005    /**
006     * Event that indicates a crash
007     */
008    public class CrashEvent extends Event<CrashEvent> {
009            
010            Vehicle[] vehicles;
011            
012            /**
013             * Initialise
014             * 
015             * @param timestamp
016             * @param target
017             */
018            public CrashEvent(long timestamp, IEventTarget<CrashEvent> target,Vehicle one, Vehicle other) {
019                    super(timestamp, target);
020                    vehicles = new Vehicle[]{one,other};
021            }
022    
023            @Override
024            public void handleEvent() throws Exception {
025                    this.target.handleEvent(this);
026            }
027            
028            /**
029             * Return the vehicles that have crashed
030             * @return Crashed vehicles
031             */
032            public Vehicle[] getVehicles(){
033                    return vehicles;
034            }
035    }
