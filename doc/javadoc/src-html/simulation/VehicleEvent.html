001    package simulation;
002    
003    /**
004     * Event for the vehicles
005     */
006    public class VehicleEvent extends Event<VehicleEvent> {
007            /**
008             * The acceleration to set on the vehicle
009             */
010            private float targetAcceleration = 0;
011    
012            /**
013             * Initialise
014             * @param timestamp The time the event occurred
015             * @param target The target of the event
016             */
017            public VehicleEvent(long timestamp, IEventTarget<VehicleEvent> target) {
018                    super(timestamp, target);
019            }
020    
021            /**
022             * Initialise
023             * @param timestamp The time the event occurred
024             * @param target The target of the event
025             * @param targetAcceleration The target acceleration the car should use
026             */
027            public VehicleEvent(long timestamp, IEventTarget<VehicleEvent> target,
028                            float targetAcceleration) {
029                    super(timestamp, target);
030    
031                    this.targetAcceleration = targetAcceleration;
032            }
033    
034            /**
035             * Set the target acceleration
036             * @param targetAcceleration The acceleration the car should use
037             * @return self-reference
038             */
039            public VehicleEvent setTargetAcceleration(float targetAcceleration) {
040                    this.targetAcceleration = targetAcceleration;
041    
042                    return this;
043            }
044    
045            /**
046             * Return the target acceleration
047             * @return the target acceleration
048             */
049            public float getTargetAcceleration() {
050                    return this.targetAcceleration;
051            }
052    
053            @Override
054            public void handleEvent() throws Exception {
055                    this.target.handleEvent(this);
056            }
057    }
