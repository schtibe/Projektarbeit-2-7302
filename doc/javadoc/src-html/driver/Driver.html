001    package driver;
002    
003    import simulation.DriverEvent;
004    import simulation.EventQueue;
005    import car.IVehicle;
006    import car.Vehicle;
007    import environment.IJunctionDecision;
008    
009    public class Driver implements IDriver {
010            
011            protected IVehicle vehicle;
012            protected Animus animus;
013            
014            protected IJunctionDecision junctionDecision = null;
015            
016            @Deprecated
017            public Driver (IVehicle vehicle){
018                    this.vehicle = vehicle;
019            }
020            
021            public Driver (IVehicle vehicle, Character character, Physics physics){
022                    this.vehicle = vehicle;
023                    this.animus = new Animus (physics, character);
024                    this.animus.setVehicle ((Vehicle) vehicle);
025            }
026            
027            /**
028             * get the drivers animus
029             * @return
030             */
031            
032            public Animus getAnimus() {
033                    return this.animus;
034            }
035            
036            /* (non-Javadoc)
037             * @see driver.IDriver#handleEvent(simulation.DriverEvent)
038             */
039            
040            /**
041             * handles a driver event
042             */
043            
044            @Override
045            public void handleEvent(DriverEvent event) throws Exception{
046            
047                    if (event.getMonitoring()) {
048                            //DriverView carView = this.vehicle.getDriverView();
049                            this.animus.assessSituation(event);
050                            
051                            EventQueue.getInstance().addEvent(new DriverEvent(
052                                            event.getTimeStamp()+this.animus.getPhysics().getUpdateInterval(),
053                                            this
054                            ).setMonitoring(true));
055                    }
056                    
057                    if (event.getDecision() != null) {
058                            this.junctionDecision = event.getDecision();
059                            this.animus.setDecision(this.junctionDecision);
060                            try {
061                                    this.vehicle.setLanes(event.getDecision().getLanes());
062                            } catch (IllegalStateException e) {}
063                    }
064            }
065    }
