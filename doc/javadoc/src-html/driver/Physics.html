001    package driver;
002    
003    import java.util.List;
004    
005    import common.GlobalConstants;
006    
007    /**
008     * The physics properties of a driver
009     */
010    public class Physics {
011            
012            private float sight;
013            private float fieldOfView;
014            private int updateInterval;
015            private List<Drug> drugs;
016            
017            /**
018             * constructor for physics object
019             * @param sight How far the driver can see
020             * @param fov The angle of the view
021             * @param interval The interval to assess the situation
022             * @param drugs Drugs that are taken
023             */
024            public Physics (float sight, float fov, int interval, List<Drug> drugs){
025                    this.sight = sight * GlobalConstants.getInstance().getScale();
026                    this.fieldOfView = fov;
027                    this.updateInterval = interval;
028                    this.drugs = drugs;
029                    for (Drug drug : this.drugs){
030                            this.sight += drug.sightEffects();
031                            this.fieldOfView += drug.fieldOfViewEffects();
032                            this.updateInterval += drug.intervalEffects();
033                    }
034            }
035            
036            /**
037             * extends the driver view by its physics components
038             * @param view
039             * @return The adjusted view
040             */
041            public IDriverView getView(IDriverView view){
042                    view.setAngle(this.fieldOfView); 
043                    view.setDistance(this.sight);
044                    return view;
045            }
046            
047            /**
048             * returns the interval value
049             * @return Interval
050             */
051            public int getUpdateInterval (){
052                    return this.updateInterval;
053            }
054    }
