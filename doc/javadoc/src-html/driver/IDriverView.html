001    package driver;
002    
003    import common.IRectangle;
004    import common.IVector;
005    
006    /**
007     * The interface for driver views
008     */
009    public interface IDriverView {
010    
011            /**
012             * sets the angle
013             * @param value The view Angle
014             */
015            public abstract void setAngle(float value);
016    
017            /**
018             * Sets the distance
019             * @param value
020             */
021            public abstract void setDistance(float value);
022    
023            /**
024             * Sets the direction vector
025             * @param value The direction of the view
026             */
027            public abstract void setDirection(IVector value);
028    
029            /**
030             * Sets the position
031             * @param value Position of the vector
032             */
033            public abstract void setPosition(IVector value);
034    
035            /**
036             * Get the angle
037             * @return The angle
038             */
039            public abstract float getAngle();
040    
041            /**
042             * get the distance
043             * @return Distance
044             */
045            public abstract float getDistance();
046    
047            /**
048             * Get the position
049             * @return Position
050             */
051            public abstract IVector getPosition();
052    
053            /**
054             * Get the direction vector
055             * @return Direction
056             */
057            public abstract IVector getDirection();
058            
059            /**
060             * Returns a clone of the actual DriverView
061             * @return The clone
062             */
063            public abstract IDriverView clone();
064    
065            @Override
066            public String toString();
067            
068            /**
069             * Check whether the way point lies in this area
070             * @param position
071             * @return Wheter it lies in the arey
072             */
073            public boolean checkWayPoint(IVector position);
074            
075            /**
076             * Return a Rectangle that represents the bounding box of the DriverView
077             */
078            public IRectangle getBoundingBox();
079    }
