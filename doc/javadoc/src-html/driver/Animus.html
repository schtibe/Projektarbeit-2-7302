082            /**
083             * Method that simulates the drivers mind
084             * 
085             * This method is the heart of the project. It collects the way points
086             * of the driver view and processes them, takes the decision.
087             * @param vehicle
088             * @param event
089             * @TODO refactor if you dare ;)
090             * @throws Exception
091             */
092            public void assessSituation (DriverEvent event) throws Exception{
093                    if (vehicle.isFreezed()){
094                            return;
095                    }
096                    
097                    this.forgetJunction();
098                    
099                    this.event = event;
100                    this.noVehicles = true;
101                    // the vehicle and the junction can only influence the deceleration
102                    DecelerationActivator vehicleActivator = new DecelerationActivator(0f);
103                    DecelerationActivator junctionActivator = new DecelerationActivator(0f);
104                    
105                    IDriverView dView = this.physics.getView(vehicle.getDriverView());
106                    List<IPlacable> wayPoints = WayPointManager.getInstance().findWayPoints(dView);
107                    vehicleWayPoints = new ArrayList<VehicleWayPoint>();
108                    for(IPlacable waypoint : wayPoints) {
109                            ((IWayPoint)waypoint).visitHandleWayPoint(this);
110                    }
111                    
112                    if (isOnJunction()){
113                            List<ILane> lanes = this.currentJunction.getJunction().getRelevantLanes(this.vehicle.getLane());
114                            float nextWaypoint = Float.MAX_VALUE;
115                            for (VehicleWayPoint waypoint:vehicleWayPoints){
116                                    List<ILane> wpLanes = new ArrayList<ILane>(waypoint.getVehicle().getLanes());
117                                    for (ILane lane : wpLanes){
118                                            if (lanes.contains(lane)){
119                                                    Vehicle vehicle = waypoint.getVehicle();
120                                                    if (this.crossingVehicles.keySet().contains(waypoint)){
121                                                            if (this.crossingVehicles.get(waypoint)){
122                                                                    System.out.println(this.vehicle.hashCode()+" knows already "+vehicle.hashCode());
123                                                                    float distance = waypoint.getDistance(this.vehicle);
124                                                                    if (distance < nextWaypoint){
125                                                                            nextWaypoint = distance;
126                                                                    }
127                                                            }
128                                                    }else{
129                                                            System.out.println(this.vehicle.hashCode()+" vehicle is on my lanes "+vehicle.hashCode());
130                                                            if (this.decision != null){
131                                                                    IDirection dir = this.decision.getDirection();
132                                                                    //TODO remember each car u saw that can cross your trail and handle that cars position before u evaluate
133                                                                    //System.out.println("coming from: ");
134                                                                    IDirection from = this.currentJunction.getJunction().comingFrom(this.vehicle.getLane(), vehicle.getLane());
135                                                                    IDirection to = vehicle.getSimpleDirection();
136                                                                    //System.out.println("d:"+dir.toString()+";f:"+from.toString()+";t:"+to.toString());
137                                                                    if (dir.crossesMe(from,to)){
138                                                                            this.crossingVehicles.put(waypoint, true);
139                                                                            //System.out.println(this.vehicle.hashCode()+" has crossing "+vehicle.hashCode());
140                                                                            IPriority priority = new PriorityRight();
141                                                                            boolean prio = priority.hasPriority(dir, from, to);
142                                                                            //System.out.println(prio);
143                                                                            if (prio){
144                                                                                    System.out.println(vehicle.hashCode()+"has priority over"+this.vehicle.hashCode());
145                                                                                    float distance = waypoint.getDistance(this.vehicle);
146                                                                                    if (distance < nextWaypoint){
147                                                                                            nextWaypoint = distance;
148                                                                                    }
149                                                                                    this.crossingVehicles.put(waypoint, true);
150                                                                            }else{
151                                                                                    this.crossingVehicles.put(waypoint, false);
152                                                                            }
153                                                                    }
154                                                                    break;
155                                                            }
156                                                    }
157                                            }
158                                    }
159                            }
160                            if (nextWaypoint < securityDistance() && nextWaypoint < nextWaypointOld){
161                                    junctionActivator.setValue(
162                                                    this.calculateVehicleActivator(
163                                                                    securityDistance(),
164                                                                    nextWaypoint,
165                                                                    this.vehicle.getSpeed()
166                                                    )
167                                    );
168                                    nextWaypointOld = nextWaypoint;
169                            }
170                    }
171                    
172                    if (this.noVehicles) {
173                            nearestVehicleDistanceOld = Float.MAX_VALUE;
174                            nearestVehicleDistance = Float.MAX_VALUE;
175                    }
176                    
177                    float securityDistance = this.securityDistance();
178                    if (nearestVehicleDistance < securityDistance) {
179                            vehicleActivator.setValue(
180                                            /*
181                                            this.calculateVehicleActivator(
182                                                            
183                                            )*/
184                                            this.calculateVehicleActivator(
185                                                            securityDistance(),
186                                                            nearestVehicleDistance,
187                                                            this.vehicle.getSpeed()
188                                            )
189                            );
190                    }
191                    
192                    // save the distance of the nearest vehicle for the next assessment
193                    nearestVehicleDistanceOld = nearestVehicleDistance;
194                    
195                    float acceleration = 0;
196                    float speedAssessed = assessSpeeds(this.vehicle.getSpeed(),targetSpeed);
197                    
198                    // calculate the acceleration depending on the activators
199                    if (this.vehicle.getSpeed() > targetSpeed) {
200                            acceleration = generateAcceleration(
201                                            new DecelerationActivator(speedAssessed), 
202                                            vehicleActivator, 
203                                            junctionActivator
204                            );
205                    } else {
206                            acceleration = generateAcceleration(
207                                            new AccelerationActivator(speedAssessed),
208                                            vehicleActivator,
209                                            junctionActivator
210                            );
211                    }
212                    
213                    VehicleEvent evt = new VehicleEvent(
214                                    event.getTimeStamp() + physics.getUpdateInterval(),
215                                    vehicle,
216                                    acceleration
217                    );
218                    EventQueue.getInstance().addEvent(evt);
219            }

221            /**
222             * Returns whether the speed should be changed to achieve the target speed
223             * 
224             * @param vehicle
225             * @param target
226             * @return Decision about changing the speed
227             */
228            private float assessSpeeds(float vehicle, float target) {
229                    float percentage = Math.abs((vehicle/target)-1f);
230                    
231                    if (percentage < GlobalConstants.getInstance().getSpeedPerceptionThreshold()) {
232                            return 0;
233                    }
234                    
235                    if (percentage > GlobalConstants.getInstance().getAccelerationThreshold()) {
236                            return 1f;
237                    } else { 
238                            return (1f/GlobalConstants.getInstance().getAccelerationThreshold()) * percentage;
239                    }
240            }

243             * Bring the activators together
244             * 
245             * @param speedActivator
246             * @param vehicleActivator
247             * @param junctionActivator
248             * @return The calculated acceleration
249             */
250            private float generateAcceleration(DecelerationActivator speedActivator,
251                            DecelerationActivator vehicleActivator,
252                            DecelerationActivator junctionActivator) {
253                    return calculateAcceleration (
254                                    -character.changeActivator(vehicleActivator).getValue(), 
255                                    -character.changeActivator(junctionActivator).getValue(),
256                                    -character.changeActivator(speedActivator).getValue()
257                    );

278            /**
279             * Calculate, how heavy the driver should step on the brakes
280             * 
281             * @param vehicleDistance The distance to the next vehicle
282             * @param speed The current speed
283             * @return The calculated activator value
284             */
285            private float calculateVehicleActivator(
286                            float securityDistance,
287                            float vehicleDistance, 
288                            float speed
289                    ) {
290                    float brakeWay = (float) Math.pow((speed / 10), 2);
291                    float map = securityDistance - brakeWay;
292                    float dist = vehicleDistance - brakeWay;
293                    
294                    float res = 1 - (dist / map);
295                    
296                    return res;
297            }

307            /**
308             * Calculate the acceleration from the activators
309             * 
310             * @param vehicle The activator value from the vehicle way points
311             * @param junction The activator value from the junction way points
312             * @param speed The activator value from the speed way points
313             * @return The resulting acceleration
314             */
315            private float calculateAcceleration (float vehicle, float junction, float speed) {
316                    GlobalConstants constants = GlobalConstants.getInstance();
317                    float acceleration;
318                    float n;
319                    float weightN = constants.getJunctionWaypointInfluence()+constants.getVehicleWaypointInfluence();
320                    if (vehicle == 0 || junction == 0){
321                            n = vehicle+junction;
322                    } else {
323                            n = (vehicle*constants.getVehicleWaypointInfluence()+junction*constants.getJunctionWaypointInfluence())/
324                            (weightN); 
325                    }
326                    
327                    if (n == 0 || speed == 0) {
328                            acceleration = n+speed;
329                    } else {
330                            acceleration = (n*weightN+speed*constants.getSpeedWaypointInfluence())/(weightN+constants.getSpeedWaypointInfluence());
331                    }
332                    
333                    return acceleration;
334            }

357             * handles a junction way point
358             * @param waypoint
359             */
360            public void handleWayPoint (JunctionWayPoint waypoint) {
361                    this.rememberJunction(waypoint);
362                    if (this.vehicle.getLanes().size() < Vehicle.queueSize) {
363                            if (this.checkWayPoint(vehicle, waypoint)) {
364                                    List<IJunctionDecision> decisions = 
365                                            waypoint.getJunction().getPossibilities(waypoint.getLane());
366                                    IJunctionDecision decision = decisions.get((int)Math.round(Math.random()*(decisions.size()-1)));
367                                    EventQueue.getInstance().addEvent(
368                                                    new DriverEvent(this.event.getTimeStamp(), 
369                                                    this.event.getTarget()).setDecision(decision));
370                                    
371                                    this.vehicle.notify("signal " + decision.getDirection().toString());
372                            }
373                    }
374            }

408            /**
409             * Handle a car
410             * @param carWayPoint
411             */
412            public void handleWayPoint(VehicleWayPoint waypoint) {
413                    if (waypoint != (VehicleWayPoint)vehicle.getWayPoint()){
414                            vehicleWayPoints.add(waypoint);
415                            VehicleDimension myDim = this.vehicle.getDimension();
416                            if (this.vehicle.getLanes().contains(waypoint.getLane())){
417                                    float distance = waypoint.getDistance(this.vehicle);
418                                    if (nearestVehicleDistance > distance){
419                                            nearestVehicleDistance = distance;      
420                                    } else if (noVehicles) {
421                                            // we are the first vehicle so set the distance to it
422                                            nearestVehicleDistance = distance;
423                                    }
424                                    VehicleDimension otherDim = waypoint.getVehicle().getDimension();
425                                    if (distance < myDim.getBoundingRadius()+otherDim.getBoundingRadius()){
426                                            detectCollison(waypoint);
427                                    }
428                            }
429                    }
430                    noVehicles = false;
431            }

433            /**
434             * @param waypoint
435             */
436            private void detectCollison(VehicleWayPoint waypoint) {
437                    IVector LengthA = this.vehicle.getDirection().normalize().multiply(vehicle.getDimension().getLength());
438                    IVector WidthA = this.vehicle.getDirection().normalize().multiply(vehicle.getDimension().getWidth()).rotate((float)Math.PI/2);
439                    
440                    IVector LengthB = waypoint.getVehicle().getDirection().normalize().multiply(vehicle.getDimension().getLength());
441                    IVector WidthB = waypoint.getVehicle().getDirection().normalize().multiply(vehicle.getDimension().getWidth()).rotate((float)Math.PI/2);
442                    
443                    IVector lowerLeftA = vehicle.getPosition().sub(LengthA.multiply(0.5f)).sub(WidthA.multiply(0.5f));
444                    IVector lowerLeftB = waypoint.getVehicle().getPosition().sub(LengthB.multiply(0.5f)).sub(WidthB.multiply(0.5f));
445                    
446                    boolean testedA = checkInside(LengthA, WidthA, lowerLeftA,vehicle.getPosition());
447                    
448                    boolean testedB = checkInside(LengthB, WidthB, lowerLeftB,waypoint.getVehicle().getPosition());
449                    
450                    if (testedB || testedA){
451                            CrashEvent crash = new CrashEvent(
452                                            1,
453                                            Simulator.getInstance(),
454                                            this.vehicle,
455                                            waypoint.getVehicle());
456                            EventQueue.getInstance().addEvent(crash);
457                    }
458            }

460            /**
461             * Check if a rectangle intersects
462             * 
463             * @param length The length of the rectangle to check
464             * @param width The width of the rectangle to check
465             * @param corner The bottom left (?) corner of the rectangle
466             * @param center The center
467             * @return Wheter the rectangle intersect
468             */
469            private boolean checkInside(IVector length, IVector width,
470                            IVector corner, IVector center) {
471                    IVector[] toTest = new IVector[4];
472                    
473                    toTest[0] = corner;
474                    toTest[1] = corner.add(length);
475                    toTest[2] = corner.add(width);
476                    toTest[3] = toTest[2].add(length);
477    
478                    for (IVector testV : toTest){
479                            LinearCombination comb = Vector.getLinearCombination (center,length,width,testV);
480                            if (
481                                    comb.getMu() > 0 &&
482                                    comb.getMu() < 1 &&
483                                    comb.getLambda() > 0 &&
484                                    comb.getLambda() < 1
485                            ){
486                                    return true;
487                            }
488                    }
489                    return false;
490            }

508            /**
509             * Check if a junction way point has already been seen
510             * 
511             * Do not use this for other way points than junction way points
512             * checks if a way point is important for this lane
513             * Only be interested in junction way points that lie on the
514             * current lane.
515             * 
516             * @param vehicle The vehicle for having the lanes
517             * @param waypoint The waypoint to check
518             */
519            protected boolean checkWayPoint(IVehicle vehicle, IWayPoint waypoint)  {
520                            if (vehicle.getLanes().peek() == waypoint.getLane()) { // only use the current lane
521                                    // test if it's already seen
522                                    boolean seen = false;
523                                    for (IWayPoint wp: this.seenWayPoints) {
524                                            if (waypoint == wp) {
525                                                    seen = true;
526                                                    break;
527                                            }
528                                    }
529                                    if (!seen) {
530                                            this.seenWayPoints.add(waypoint);
531                                            return true;
532                                    }
533                            }
534                    return false;
535            }

537            /**
538             * Remove the way points that are not on the lane
539             */
540            private void laneChange(ILane lastLane) {
541                    for (IWayPoint wp: this.seenWayPoints) {
542                            if (wp.getLane() == lastLane) {
543                                    this.seenWayPoints.remove(wp);
544                            }
545                    }
546            }
