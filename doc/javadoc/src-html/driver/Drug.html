001    package driver;
002    
003    /**
004     * Simulate a drug intake 
005     */
006    public abstract class Drug {
007            
008            /**
009             * prepared class can be used later to influence character and physics
010             */     
011            protected int effectOnInterval;
012            protected float effectOnSight;
013            protected float effectOnFieldOfView;
014            
015            /**
016             * The effect on the reaction time
017             * @return Interval effect
018             */
019            public int intervalEffects(){
020                    return effectOnInterval;
021            }
022            
023            /**
024             * The effect on the sight
025             * @return Sight effect
026             */
027            public float sightEffects(){
028                    return effectOnSight;
029            }
030            
031            /**
032             * The effect on angle of the view
033             * @return View angle effect
034             */
035            public float fieldOfViewEffects() {
036                    return effectOnFieldOfView;
037            }       
038    }
