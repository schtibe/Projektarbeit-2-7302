001    package common;
002    
003    /**
004     * Interface for objects that can be observed
005     */
006    public interface IObservable {
007            /**
008             * Register a new observer
009             * @param obs The observer to register
010             */
011            public void register (IObserver obs);
012            
013            /**
014             * Remove an observer
015             * @param obs The observer to remove
016             */
017            public void remove (IObserver obs);
018            
019            /**
020             * Notify all observers 
021             * @param msg The notification message
022             */
023            public void notify (String msg);
024    }
