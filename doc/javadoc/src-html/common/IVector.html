001    package common;
002    
003    /**
004     * Interface for vectors
005     */
006    public interface IVector extends Comparable<IVector> {
007            /**
008             * adding two vectors
009             * 
010             * @param the vector to be added to this vector
011             * @return a vector as a result of the operation
012             */
013            public IVector add(IVector vector);
014    
015            /**
016             * subtracting two vectors
017             * 
018             * @param the vector to be subtracted from this vector
019             * @return a vector as a result of the operation
020             */
021            public IVector sub(IVector vector);
022    
023            /**
024             * the norm of this vector
025             * 
026             * @return The length of the vector
027             */
028            public float norm();
029    
030            /**
031             * the cross product of two vectors if defined
032             * 
033             * @param the vector to calculate the cross product with this vector
034             * @return a vector as a result of the operation
035             */
036            public IVector cross(IVector vector);
037    
038            /**
039             * the dot product of two vectors
040             * 
041             * @param the vector to calculate the dotproduct with this vector
042             * @return a float as a result of the operation
043             */
044            public float dot(IVector vector);
045    
046            /**
047             * returns the component at index [0,1,..]
048             * 
049             * @param index of a component of the vector
050             * @return the components value
051             */
052            public float getComponent(int index);
053    
054            /**
055             * normalise the vector
056             * 
057             * @return a vector with length 1.0 in the same direction
058             */
059            public IVector normalize();
060    
061            /**
062             * returns a duplicate of this vector
063             * 
064             * @return
065             */
066            public IVector clone();
067    
068            /**
069             * returns a multiple of the factor (stretched)
070             * 
071             * @param factor to multiply this vector with
072             * @return a scaled copy of the actual vector
073             */
074            public IVector multiply(float factor);
075    
076            /**
077             * returns x: value, y:value
078             * 
079             * @return
080             */
081            @Override
082            public String toString();
083    
084            /**
085             * returns the vector rotated by angle, angle is measured in radians
086             * 
087             * @param an angle in the radiant notation [0,2*PI]
088             * @return a vector rotated by the input angle
089             */
090            public IVector rotate(float angle);
091    
092            /**
093             * the result is the angle between this vector and the x axis in radians
094             * 
095             * @return an angle in radiant
096             */
097            public float getAngle();
098    
099    }
