001    package common;
002    
003    public class Rectangle implements IRectangle {
004    
005            private IVector bottomLeft;
006            private IVector topRight;
007            private float width;
008            private float height;
009            
010            public Rectangle (IVector botleft, IVector topright){
011                    bottomLeft = botleft;
012                    topRight = topright;
013            }
014            
015            @Override
016            public IVector getBottomLeft() {
017                    return bottomLeft;
018            }
019    
020            @Override
021            public IVector getTopRight() {
022                    return topRight;
023            }
024    
025            @Override
026            public float getWidth() {
027                    if (width == 0){
028                            width = topRight.getComponent(0)-bottomLeft.getComponent(0);
029                            
030                    }
031                    return width;
032            }
033    
034            @Override
035            public float getHeight() {
036                    if (height == 0){
037                            height = topRight.getComponent(1)-bottomLeft.getComponent(1);
038                    }
039                    return height;
040            }
041            
042            @Override
043            public String toString (){
044                    return bottomLeft.toString()+":"+topRight.toString();
045            }
046    
047    }
