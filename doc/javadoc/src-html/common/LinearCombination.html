001    package common;
002    
003    /**
004     * The result of a linear combination
005     */
006    public class LinearCombination {
007            private float mu;
008            private float lambda;
009            
010            public LinearCombination (float m, float l){
011                    mu = m;
012                    lambda = l;
013            }
014            
015            /**
016             * Get the mu value of the linear combination
017             * @return Mu
018             */
019            public float getMu (){
020                    return mu;
021            }
022            
023            /**
024             * Get the lambda value of the linear combination
025             * @return Lambda
026             */
027            public float getLambda(){
028                    return lambda;
029            }
030    }
