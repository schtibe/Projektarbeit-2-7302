001    package common;
002    
003    /**
004     * implementation of a 2D Vector
005     */
006    public class Vector implements IVector {
007    
008            /**
009             * The vectors components
010             */
011            private float[] components;
012    
013            /**
014             * Is the vectors length already calculated
015             */
016            private boolean lengthCalculated = false;
017    
018            /**
019             * The vectors length
020             */
021            private float length;
022    
023            /**
024             * @param array of floats, the vectors components
025             */
026            public Vector(float[] components) {
027                    if (components != null) {
028                            if (components.length == 2) {
029                                    this.components = components.clone();
030                            } else {
031                                    this.components = new float[2];
032                                    int i;
033                                    for (i = 0; i < components.length; i++) {
034                                            if (i < 2) {
035                                                    this.components[i] = components[i];
036                                            } else {
037                                                    break;
038                                            }
039                                    }
040                                    if (i < 2) {
041                                            for (int j = i; j < 2; j++) {
042                                                    this.components[j] = 0;
043                                            }
044                                    }
045                            }
046                    } else {
047                            this.components = new float[] { 0, 0 };
048                    }
049            }
050    
051            @Override
052            public Vector add(IVector vector) {
053                    return new Vector(new float[] { components[0] + vector.getComponent(0),
054                                    components[1] + vector.getComponent(1) });
055            }
056    
057            /**
058             * {@inheritDoc}
059             * 
060             *         the cross product of 2 2d vectors is usually (0,0,z) the
061             *         corresponding 2d vector is (0,0) hence the function will always
062             *         return (0,0)
063             */
064            @Override
065            public Vector cross(IVector vector) {
066                    return new Vector(new float[] { 0, 0 });
067            }
068    
069            @Override
070            public float dot(IVector vector) {
071                    float scalar = 0;
072                    for (int i = 0; i < this.components.length; i++) {
073                            scalar += this.components[i] * vector.getComponent(i);
074                    }
075                    return scalar;
076            }
077    
078    
079            @Override
080            public float getComponent(int index) {
081                    if (index >= this.components.length) {
082                            return 0;
083                    } else {
084                            return this.components[index];
085                    }
086            }
087    
088            @Override
089            public float norm() {
090                    if (!lengthCalculated) {
091                            for (int i = 0; i < this.components.length; i++) {
092                                    length += this.components[i] * this.components[i];
093                            }
094                            this.length = (float) Math.sqrt(length);
095                            lengthCalculated = true;
096                    }
097                    return this.length;
098            }
099    
100            @Override
101            public Vector sub(IVector vector) {
102                    return new Vector(new float[] { components[0] - vector.getComponent(0),
103                                    components[1] - vector.getComponent(1) });
104            }
105    
106            @Override
107            public Vector normalize() {
108                    float[] newComponents = new float[this.components.length];
109                    for (int i = 0; i < this.components.length; i++) {
110                            newComponents[i] = this.components[i] / this.norm();
111                    }
112                    return new Vector(newComponents);
113            }
114    
115            @Override
116            public Vector clone() {
117                    return new Vector(this.components);
118            }
119    
120            @Override
121            public Vector multiply(float factor) {
122                    float[] newComponents = new float[this.components.length];
123                    for (int i = 0; i < this.components.length; i++) {
124                            newComponents[i] = this.components[i] * factor;
125                    }
126                    return new Vector(newComponents);
127            }
128    
129            @Override
130            public String toString() {
131                    return "x: " + this.components[0] + ", y: " + this.components[1];
132            }
133    
134            @Override
135            public Vector rotate(float angle) {
136                    float newAngle = this.getAngle() + angle;
137                    return new Vector(new float[] {
138                                    (float) Math.cos(newAngle) * this.norm(),
139                                    (float) Math.sin(newAngle) * this.norm() });
140            }
141    
142            /**
143             * Return the angle to the X axis 
144    
145             */
146            @Override
147            public float getAngle() {
148                    IVector newThis = this.normalize();
149                    IVector xAxis = new Vector(new float[] { 1, 0 });
150                    float cosAlpha = newThis.dot(xAxis);
151                    float direction = (float) Math.atan2(newThis.getComponent(1), newThis
152                                    .getComponent(0));
153                    float angle = (float) Math.acos(cosAlpha);
154                    if (direction >= 0) {
155                            return angle;
156                    } else {
157                            return (float) (Math.PI * 2) - angle;
158                    }
159            }
160    
161            @Override
162            public int compareTo(IVector arg0) {
163                    Boolean isEqual = true;
164                    for (int i = 0; i < this.components.length; i++) {
165                            if (this.components[i] != arg0.getComponent(i)) {
166                                    isEqual = false;
167                            }
168                    }
169                    if (isEqual) {
170                            return 0;
171                    } else if (this.norm() > arg0.norm()) {
172                            return 1;
173                    } else {
174                            return -1;
175                    }
176            }
177            
178            /**
179             * Return the result of a linear combination
180             * 
181             * @param center 
182             * @param directionA
183             * @param directionB
184             * @param point
185             * @return The result of the linear combination
186             */
187            public static LinearCombination getLinearCombination (IVector center, IVector directionA, IVector directionB, IVector point){
188                    IVector position = point.sub(center);   
189                    float denominator = 
190                            directionA.getComponent(0) * directionB.getComponent(1)
191                            - directionA.getComponent(1) * directionB.getComponent(0);
192                    
193                    float lambda = -(directionB.getComponent(0) * position.getComponent(1)
194                                    - directionB.getComponent(1) * position.getComponent(0)) /
195                                    denominator;
196                    
197                    float mu = (directionA.getComponent(0) * position.getComponent(1)
198                                    - directionA.getComponent(1) * position.getComponent(0)) /
199                                    denominator;
200                    
201                    return new LinearCombination (mu, lambda);
202            }
203    }
