001    package car;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    import java.util.Queue;
006    import java.util.concurrent.ArrayBlockingQueue;
007    
008    import simulation.VehicleEvent;
009    
010    import common.GlobalConstants;
011    import common.IObserver;
012    import common.IVector;
013    import common.Vector;
014    
015    import driver.DriverViewCircleSector;
016    import driver.IDirection;
017    import driver.IDriverView;
018    import environment.ILane;
019    import environment.IMovable;
020    import environment.LaneLengthExceededException;
021    import environment.WayPointManager;
022    
023    /**
024     * The abstract vehicle
025     * 
026     * This class contains all the needed information and methods for a vehicle.
027     * 
028     */
029    public abstract class Vehicle implements IVehicle {
030    
031            /**
032             * List of Observers, that get notified when a change occurs
033             */
034            private List<IObserver> observers;
035    
036            /**
037             * Indicate that the vehicle has crashed
038             */
039            protected boolean freezed;
040    
041            /**
042             * The way point that belongs to this vehicle
043             */
044            protected IMovable wayPoint;
045    
046            /**
047             * The direction the vehicle is heading
048             * 
049             * This vector has always the length 1 (unit vector)
050             */
051            protected IVector direction;
052    
053            /**
054             * The last direction
055             */
056            protected IVector lastPosition;
057    
058            /**
059             * The current speed of the vehicle
060             */
061            protected float speed = 0;
062    
063            /**
064             * The position in the simulation of the vehicle
065             */
066            protected IVector position;
067    
068            public final static int queueSize = 12;
069    
070            /**
071             * The vehicle's dimension
072             */
073            protected VehicleDimension dimension;
074    
075            /**
076             * The driver's view
077             * 
078             * @TODO probably move this to another place
079             */
080            protected IDriverView driverView;
081    
082            @Override
083            public VehicleDimension getDimension() {
084                    return dimension;
085            }
086    
087            @Override
088            public void setDimension(VehicleDimension dimension) {
089                    this.dimension = dimension;
090            }
091    
092            /**
093             * The current acceleration
094             * 
095             * This acceleration is added to the speed and/or direction
096             */
097            protected float acceleration = 0;
098    
099            /**
100             * The lane the vehicle is on
101             */
102            protected Queue<ILane> lanes = new ArrayBlockingQueue<ILane>(
103                            Vehicle.queueSize);
104    
105            /**
106             * The driven distance on the lane
107             */
108            protected float drivenLaneDistance = 0;
109    
110            private IDirection dir;
111    
112            @Override
113            public float getSpeed() {
114                    return this.speed;
115            }
116    
117            /**
118             * Construct
119             * 
120             * @param lane The lane the vehicle is on
121             */
122            public Vehicle(ILane lane) throws IllegalArgumentException {
123                    if (lane == null) {
124                            throw new IllegalArgumentException();
125                    }
126    
127                    this.construct(lane, 0);
128            }
129    
130            /**
131             * Construct
132             * 
133             * Create a car and position it on a lane
134             * 
135             * @param lane The lane the car is on
136             * @param drivenLaneDistance The position on the lane
137             */
138            public Vehicle(ILane lane, float drivenLaneDistance)
139                            throws IllegalArgumentException {
140                    if (lane == null) {
141                            throw new IllegalArgumentException();
142                    }
143    
144                    this.construct(lane, drivenLaneDistance);
145            }
146    
147            /**
148             * Initialise all needed variables
149             * 
150             * @param lane
151             * @param drivenLaneDistance
152             */
153            private void construct(ILane lane, float drivenLaneDistance) {
154                    this.lanes.add(lane);
155                    this.drivenLaneDistance = drivenLaneDistance;
156                    try {
157                            this.position = this.lanes.peek().getPositionOnLane(
158                                            this.drivenLaneDistance);
159                    } catch (LaneLengthExceededException e) {
160                            try {
161                                    this.position = this.lanes.peek().getPositionOnLane(0);
162                            } catch (LaneLengthExceededException e1) {
163                                    e1.printStackTrace();
164                            }
165                    }
166                    this.initializeDirection();
167                    this.createWayPoint();
168    
169                    this.driverView = new DriverViewCircleSector(this.direction.normalize(),
170                                    this.position);
171            }
172    
173            /**
174             * Set the direction upon initialisation
175             */
176            private void initializeDirection() {
177                    this.direction = this.lanes.peek().getFirstILaneSegment().getEndPoint()
178                                    .sub(this.lanes.peek().getFirstILaneSegment().getStartPoint());
179                    this.lastPosition = this.lanes.peek().getFirstILaneSegment()
180                                    .getStartPoint();
181            }
182    
183            /**
184             * Calculate the real acceleration of the vehicle
185             * 
186             * @param Acceleration to calculate
187             * @return Resulting acceleration
188             */
189            protected float calculateAcceleration(float acceleration) {
190                    if (acceleration > 0) {
191                            return this.maxAcceleration() * acceleration;
192                    } else {
193                            return this.maxDeceleration() * acceleration;
194                    }
195            }
196    
197            /**
198             * Return the maximal acceleration
199             * 
200             * @return Maximal acceleration
201             */
202            protected abstract float maxAcceleration();
203    
204            /**
205             * Return the maximal deceleration
206             * 
207             * @returnm Maximal deceleration
208             */
209            protected abstract float maxDeceleration();
210    
211            @Override
212            public void accelerate(float acceleration) {
213                    this.acceleration = this.calculateAcceleration(acceleration);
214            }
215    
216            @Override
217            public ILane getLane() {
218                    return this.lanes.peek();
219            }
220    
221            @Override
222            public IVector getPosition() {
223                    return new Vector(new float[] { this.position.getComponent(0),
224                                    this.position.getComponent(1) });
225            }
226    
227            @Override
228            public void updatePosition(float timestep) throws CarCannotReverseException {
229                    if (!freezed) {
230                            this.adjustSpeed(timestep);
231                            this.drivenLaneDistance += (this.speed / 36) // this seems weird?
232                                            * GlobalConstants.getInstance().getScale();
233                            try {
234                                    this.position = this.lanes.peek().getPositionOnLane(
235                                                    this.drivenLaneDistance);
236                            } catch (LaneLengthExceededException e) {
237                                    // change lane
238                                    this.drivenLaneDistance = this.drivenLaneDistance
239                                                    - this.lanes.peek().getLength();
240                                    this.notify("laneChange");
241                                    try {
242                                            this.position = this.lanes.peek().getPositionOnLane(
243                                                            this.drivenLaneDistance);
244                                    } catch (LaneLengthExceededException e1) {
245                                    } catch (NullPointerException e2) {
246                                            System.out.println("No further lane found!");
247                                            System.out.println("The car is " + this);
248                                            System.out.println("The car was on position "
249                                                            + this.position);
250                                            return;
251                                    }
252                            } catch (NullPointerException e2) {
253                                    System.out.println("No further lane found!");
254                                    System.out.println("The car is " + this);
255                                    System.out.println("The car was on position " + this.position);
256                                    return;
257                            }
258    
259                            this.driverView.setPosition(this.position);
260                            this.updateDirection();
261                            try {
262                                    WayPointManager.getInstance().move(this.wayPoint,
263                                                    this.getPosition().getComponent(0),
264                                                    this.getPosition().getComponent(1));
265                            } catch (Exception e) {
266                                    e.printStackTrace();
267                            }
268                    }
269            }
270    
271            /**
272             * Update the direction of the car
273             * 
274             * The difference vector of the old and new position is the direction the
275             * car is looking
276             */
277            public void updateDirection() {
278                    if (this.position.compareTo(this.lastPosition) == 0) {
279                            this.direction = this.lanes
280                                            .peek()
281                                            .getFirstILaneSegment()
282                                            .getEndPoint()
283                                            .sub(this.lanes.peek().getFirstILaneSegment()
284                                                            .getStartPoint());
285                    } else {
286                            this.direction = this.position.sub(this.lastPosition);
287                            this.lastPosition = this.position;
288                    }
289    
290                    this.driverView.setDirection(this.direction);
291            }
292    
293            /**
294             * Return the driven lane distance
295             * 
296             * @return Driven lane distance
297             */
298            @Override
299            public float getDrivenLaneDistance() {
300                    return this.drivenLaneDistance;
301            }
302    
303            /**
304             * Adjust the speed
305             * 
306             * @throws CarCannotReverseException
307             */
308            protected abstract void adjustSpeed(float timestep)
309                            throws CarCannotReverseException;
310    
311            /**
312             * Handle an event
313             * 
314             * @see car.IVehicle#handleEvent(simulation.IEvent)
315             * @param VehicleEvent The event to handle
316             */
317            @Override
318            public abstract void handleEvent(VehicleEvent event);
319    
320            @Override
321            public void setLane(ILane lane) {
322                    this.lanes.add(lane);
323            }
324    
325            @Override
326            public void setLanes(List<ILane> lanes) {
327                    this.lanes.addAll(lanes);
328            }
329    
330            @Override
331            public Queue<ILane> getLanes() {
332                    return this.lanes;
333            }
334    
335            @Override
336            public IVector getDirection() {
337                    return this.direction;
338            }
339    
340            @Override
341            public IDriverView getDriverView() {
342                    return this.driverView;
343            }
344    
345            @Override
346            public void register(IObserver obs) {
347                    if (observers == null) {
348                            observers = new ArrayList<IObserver>();
349                    }
350                    observers.add(obs);
351            }
352    
353            @Override
354            public void remove(IObserver obs) {
355                    observers.remove(obs);
356            }
357    
358            @Override
359            public void notify(String msg) {
360                    for (IObserver obs : observers) {
361                            obs.update(msg);
362                    }
363            }
364    
365            /**
366             * Freeze a vehicle
367             * 
368             * This is used when a vehicle crashes.s
369             */
370            public void freeze() {
371                    freezed = true;
372                    speed = 0;
373                    acceleration = 0;
374            }
375            
376            @Override
377            public boolean isFreezed(){
378                    return freezed;
379            }
380            
381            @Override
382            public void removeWaypoint(){
383                    this.wayPoint = null;
384            }
385            
386            @Override
387            public void setSimpleDirection(IDirection dir){
388                    this.dir = dir;
389            }
390            
391            @Override
392            public IDirection getSimpleDirection(){
393                    return this.dir;
394            }
395    }
